<script>
    // ==========================
    // Task (a): Change Background Color
    // ==========================
    const changeColorBtn = document.getElementById("changeColor");

    changeColorBtn.addEventListener("click", function () {
        document.body.style.backgroundColor = "#CCCCFF";
    });


    // ==========================
    // Task (b): Add List Item
    // ==========================
    const addItemBtn = document.getElementById("addItem");
    const itemList = document.getElementById("itemList");

    addItemBtn.addEventListener("click", function () {

        // Current number of list items + 1
        const itemNumber = itemList.children.length + 1;

        // Create a new <li>
        const newItem = document.createElement("li");

        // Set its class (optional, but keeps the same blue style)
        newItem.className = "item";

        // Set its text
        newItem.textContent = `Item ${itemNumber}`;

        // Add it to the list
        itemList.appendChild(newItem);

    });

    // ==========================
    // Task (c): Live Text Update
    // ==========================
    const userInput = document.getElementById("userInput");
    const output = document.getElementById("output");

    userInput.addEventListener("keyup", function () {
    output.textContent = userInput.value;
    });

</script>
