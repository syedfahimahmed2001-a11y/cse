// Binary search
------------------------------------------------------------------------------------------
#include <iostream>
using namespace std;
int main(){
    int arr[10];
    cout << "Array10(sorted): ";
    for (int i = 0; i < 10; i++)
    {
        cin >> arr[i];
    }

    int key;
    cout << "Item to search: ";
    cin >> key;
    int left = 0;
    int right = 9;
    bool found = false;
    int index;

    while (left <= right)
    {
        int mid = (left + right) / 2;

        if (arr[mid] == key)
        {
            found = true;
            index = mid;
            break;
        }
        else if(arr[mid] < key)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }
    if (found)
    {
        cout << "Found" << endl;
        cout << "Index: " << index;
    }
    else
    {
        cout << "Not Found";
    }
    
}

// Selection sort
--------------------------------------------------------------------------------------
#include <iostream>
using namespace std;
int main(){
    int arr[5];
    cout << "Enter 5 arr element: ";
    for (int i = 0; i < 5; i++)
    {
        cin >> arr[i];
    }

    for (int i = 0; i < 5; i++)
    {
        int minIndex = i;

        for (int j = i + 1; j < 5; j++)
        {
            if (arr[j] < arr[minIndex])
            {
                minIndex = j;
            }
            
        }
        int temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }

    cout << "Sorted arr: ";
    for (int i = 0; i < 5; i++)
    {
        cout << arr[i] << ", ";
    }
    
    return 0;
    
}

// Insertion sort
-------------------------------------------------------------------------------------
#include <iostream>
using namespace std;
int main(){
    int arr[5];
    cout << "Enter 5 arr element: ";
    for (int i = 0; i < 5; i++)
    {
        cin >> arr[i];
    }
    for (int i = 1; i < 5; i++)
    {
        int key = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > key)
        {
            // Right shifting
            arr[j + 1] = arr[j];
            j--;
        }
        // Insert the key in right position
        arr[j + 1] = key;
    }
    cout << "Sorted arr: ";
    for (int i = 0; i < 5; i++)
    {
        cout << arr[i];
    }
    
    return 0;
    
}

// Merge Sort
---------------------------------------------------------------------------------------
#include <iostream>
using namespace std;

// Function to merge two sorted halves
void merge(int arr[], int left, int mid, int right)
{
    // Calculate sizes of left and right subarrays
    int n1 = mid - left + 1;
    int n2 = right - mid;

    // Temporary arrays
    int L[n1];
    int R[n2];

    // Copy data into temporary arrays
    for (int i = 0; i < n1; i++)
        L[i] = arr[left + i];

    for (int j = 0; j < n2; j++)
        R[j] = arr[mid + 1 + j];

    int i = 0;       // Index for L[]
    int j = 0;       // Index for R[]
    int k = left;    // Index for original array

    // Compare elements and copy the smaller one
    while (i < n1 && j < n2)
    {
        if (L[i] <= R[j])
        {
            arr[k] = L[i];
            i++;
        }
        else
        {
            arr[k] = R[j];
            j++;
        }
        k++;
    }

    // Copy remaining elements from L[]
    while (i < n1)
    {
        arr[k] = L[i];
        i++;
        k++;
    }

    // Copy remaining elements from R[]
    while (j < n2)
    {
        arr[k] = R[j];
        j++;
        k++;
    }
}

// Recursive Merge Sort function
void mergeSort(int arr[], int left, int right)
{
    // Base case: one element
    if (left >= right)
        return;

    // Find middle index
    int mid = left + (right - left) / 2;

    // Sort left half
    mergeSort(arr, left, mid);

    // Sort right half
    mergeSort(arr, mid + 1, right);

    // Merge the two sorted halves
    merge(arr, left, mid, right);
}

int main()
{
    int arr[5];

    cout << "Enter 5 elements: ";

    for (int i = 0; i < 5; i++)
        cin >> arr[i];

    // Sort the array
    mergeSort(arr, 0, 4);

    cout << "\nSorted Array:\n";

    for (int i = 0; i < 5; i++)
        cout << arr[i] << " ";

    return 0;
}

// Quick Sort
----------------------------------------------------------------------------------------
#include <iostream>
using namespace std;

// Function to partition the array
int partition(int arr[], int low, int high)
{
    // Choose the last element as the pivot
    int pivot = arr[high];

    // Index of smaller element
    int i = low - 1;

    // Compare each element with the pivot
    for (int j = low; j < high; j++)
    {
        // If current element is smaller than or equal to the pivot
        if (arr[j] <= pivot)
        {
            i++;

            // Swap arr[i] and arr[j]
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }

    // Place the pivot in its correct position
    int temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;

    // Return the pivot index
    return i + 1;
}

// Quick Sort function
void quickSort(int arr[], int low, int high)
{
    // Base condition
    if (low < high)
    {
        // Partition the array and get the pivot index
        int pivotIndex = partition(arr, low, high);

        // Sort the left part
        quickSort(arr, low, pivotIndex - 1);

        // Sort the right part
        quickSort(arr, pivotIndex + 1, high);
    }
}

int main()
{
    int arr[5];

    cout << "Enter 5 elements: ";

    for (int i = 0; i < 5; i++)
    {
        cin >> arr[i];
    }

    // Call Quick Sort
    quickSort(arr, 0, 4);

    cout << "\nSorted Array:\n";

    for (int i = 0; i < 5; i++)
    {
        cout << arr[i] << " ";
    }

    cout << endl;

    return 0;
}

// Counting sort
-------------------------------------------------------------------------------------
#include <iostream>
using namespace std;

// Function to perform Counting Sort
void countingSort(int arr[], int n)
{
    // Step 1: Find the maximum element
    int max = arr[0];

    for (int i = 1; i < n; i++)
    {
        if (arr[i] > max)
        {
            max = arr[i];
        }
    }

    // Step 2: Create and initialize the count array
    int count[max + 1] = {0};

    // Step 3: Count the frequency of each element
    for (int i = 0; i < n; i++)
    {
        count[arr[i]]++;
    }

    // Step 4: Reconstruct the original array in sorted order
    int index = 0;

    for (int i = 0; i <= max; i++)
    {
        while (count[i] > 0)
        {
            arr[index] = i;
            index++;
            count[i]--;
        }
    }
}

int main()
{
    int n;

    cout << "Enter the number of elements: ";
    cin >> n;

    int arr[n];

    cout << "Enter the elements: ";

    for (int i = 0; i < n; i++)
    {
        cin >> arr[i];
    }

    // Call Counting Sort
    countingSort(arr, n);

    cout << "\nSorted Array:\n";

    for (int i = 0; i < n; i++)
    {
        cout << arr[i] << " ";
    }

    cout << endl;

    return 0;
}

// DFS
----------------------------------------------------------------------------------------
#include <iostream>
using namespace std;

const int MAX = 100;

// Visited array
bool visited[MAX];

// Adjacency matrix
int graph[MAX][MAX];

// Function to perform DFS
void DFS(int node, int vertices)
{
    // Mark the current node as visited
    visited[node] = true;

    // Print the current node
    cout << node << " ";

    // Visit all adjacent nodes
    for (int i = 0; i < vertices; i++)
    {
        if (graph[node][i] == 1 && !visited[i])
        {
            DFS(i, vertices);
        }
    }
}

int main()
{
    int vertices, edges;

    cout << "Enter number of vertices: ";
    cin >> vertices;

    cout << "Enter number of edges: ";
    cin >> edges;

    // Initialize adjacency matrix with 0
    for (int i = 0; i < vertices; i++)
    {
        for (int j = 0; j < vertices; j++)
        {
            graph[i][j] = 0;
        }
    }

    cout << "Enter the edges (u v):\n";

    for (int i = 0; i < edges; i++)
    {
        int u, v;
        cin >> u >> v;

        // Undirected graph
        graph[u][v] = 1;
        graph[v][u] = 1;
    }

    int start;

    cout << "Enter starting vertex: ";
    cin >> start;

    cout << "\nDFS Traversal: ";

    DFS(start, vertices);

    cout << endl;

    return 0;
}

#include <iostream>
#include <queue>
using namespace std;

const int MAX = 100;

// Adjacency matrix
int graph[MAX][MAX];

// Visited array
bool visited[MAX];

// Function to perform BFS
void BFS(int start, int vertices)
{
    // Create a queue
    queue<int> q;

    // Mark the starting vertex as visited
    visited[start] = true;

    // Push the starting vertex into the queue
    q.push(start);

    while (!q.empty())
    {
        // Remove the front vertex
        int current = q.front();
        q.pop();

        // Print the current vertex
        cout << current << " ";

        // Visit all adjacent vertices
        for (int i = 0; i < vertices; i++)
        {
            if (graph[current][i] == 1 && !visited[i])
            {
                visited[i] = true;
                q.push(i);
            }
        }
    }
}

int main()
{
    int vertices, edges;

    cout << "Enter number of vertices: ";

    {
        int u, v;
        cin >> u >> v;

        // Undirected graph
        graph[u][v] = 1;
        graph[v][u] = 1;
    }

    int start;

// Sum problem
--------------------------------------------------------------------------------------
#include <iostream>
using namespace std;

int main()
{
    int n;

    cout << "Enter the number of elements: ";
    cin >> n;

    int arr[n];

    cout << "Enter the elements: ";

    for (int i = 0; i < n; i++)
    {
        cin >> arr[i];
    }

    int target;

    cout << "Enter the target sum: ";
    cin >> target;

    // DP table
    bool dp[n + 1][target + 1];

    // Sum 0 is always possible
    for (int i = 0; i <= n; i++)
    {
        dp[i][0] = true;
    }

    // If no elements are available, positive sums are impossible
    for (int j = 1; j <= target; j++)
    {
        dp[0][j] = false;
    }

    // Fill the DP table
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= target; j++)
        {
            if (arr[i - 1] <= j)
            {
                dp[i][j] = dp[i - 1][j] || dp[i - 1][j - arr[i - 1]];
            }
            else
            {
                dp[i][j] = dp[i - 1][j];
            }
        }
    }

    // Print the result
    if (dp[n][target])
    {
        cout << "YES";
    }
    else
    {
        cout << "NO";
    }

    return 0;
}

// Beautiful String.
---------------------------------------------------------------------------------------
#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin >> t;

    while (t--)
    {
        string s;
        cin >> s;

        int n = s.length();
        bool possible = true;

        // Check if the existing string is already invalid
        for (int i = 0; i < n - 1; i++)
        {
            if (s[i] != '?' && s[i + 1] != '?' && s[i] == s[i + 1])
            {
                possible = false;
                break;
            }
        }

        if (!possible)
        {
            cout << -1 << endl;
            continue;
        }

        // Replace each '?'
        for (int i = 0; i < n; i++)
        {
            if (s[i] == '?')
            {
                for (char ch = 'a'; ch <= 'c'; ch++)
                {
                    bool ok = true;

                    // Check left neighbor
                    if (i > 0 && s[i - 1] == ch)
                        ok = false;

                    // Check right neighbor
                    if (i < n - 1 && s[i + 1] == ch)
                        ok = false;

                    if (ok)
                    {
                        s[i] = ch;
                        break;
                    }
                }
            }
        }

        cout << s << endl;
    }

    return 0;
}

// from A to B.
----------------------------------------------------------------------------------------
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    long long a, b;
    cin >> a >> b;

    vector<long long> path;

    while (b >= a)
    {
        path.push_back(b);

        if (b == a)
        {
            reverse(path.begin(), path.end());

            cout << "YES" << endl;
            cout << path.size() << endl;

            for (long long x : path)
                cout << x << " ";

            cout << endl;
            return 0;
        }

        if (b % 10 == 1)
        {
            b = (b - 1) / 10;
        }
        else if (b % 2 == 0)
        {
            b /= 2;
        }
        else
        {
            break;
        }
    }

    cout << "NO" << endl;

    return 0;
}

// String Task.
---------------------------------------------------------------------------------------
#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main()
{
    string s;
    cin >> s;

    for (int i = 0; i < s.length(); i++)
    {
        // Convert current character to lowercase
        char ch = tolower(s[i]);

        // Skip vowels
        if (ch == 'a' || ch == 'e' || ch == 'i' ||
            ch == 'o' || ch == 'u' || ch == 'y')
        {
            continue;
        }

        // Print '.' followed by the consonant
        cout << "." << ch;
    }

    cout << endl;

    return 0;
}
